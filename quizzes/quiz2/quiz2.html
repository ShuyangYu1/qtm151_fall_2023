def fn_mse(y,x,b0,b1):
    e2=(y-b0-b1*x)**2
    return e2.mean()

# Write your own code
list_status = [0,1]
list_prob = [0.5, 0.5]
n=len(Dataset)
np.random.seed(42)
Dataset["status"] = np.random.choice(list_status, size = n, p = list_prob)
data0=Dataset.query('status==0')
data1=Dataset.query('status==1')
